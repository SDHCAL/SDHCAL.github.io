66inline void Payload150::parsePayload()
67{
68 m_Frames.clear();
69 std::uint32_t fshift{static_cast<std::uint32_t>(Size::GLOBAL_HEADER)}; // Pass Global Header
70 // Pass DIF_ID, DIF Trigger counter, Information counter, Global Trigger counter, Absolute BCID, BCID DIF, NB line
71 fshift += Size::PMR_ID_SHIFT + Size::PMR_NBASIC_SHIFT + Size::PMR_FORMAT_SHIFT + Size::PMR_GTC_SHIFT + Size::PMR_ABCID_SHIFT + Size::PMR_BCID_SHIFT + Size::PMR_LTRG_SHIFT;
72 while(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::GLOBAL_TRAILER))
73 {
74 m_Frames.push_back(&begin()[fshift]);
75 fshift += Size::MICROROC_HEADER + Size::BCID + Size::DATA;
76 }
77 // Pass Global trailer
78 fshift += +Size::GLOBAL_TRAILER;
79 // Pass CRC MSB, CRC LSB
80 fshift += Size::CRC_MSB + Size::CRC_LSB;
81 theGetFramePtrReturn_ = fshift;
82}
