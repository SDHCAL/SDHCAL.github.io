68inline void Payload100::parsePayload()
69{
70 m_Frames.clear();
71 m_Lines.clear();
72 std::uint32_t fshift{static_cast<std::uint32_t>(Size::GLOBAL_HEADER)}; // Pass Global Header
73 if(m_Version >= 13)
74 {
75 // Pass DIF_ID, DIF Trigger counter, Information counter, Global Trigger counter, Absolute BCID, BCID DIF, NB line
76 fshift += Size::DIF_IF + Size::DIF_TRIGGER_COUNTER + Size::INFORMATION_COUNTER + Size::GLOBAL_TRIGGER_COUNTER + Size::ABSOLUTE_BCID + Size::BCID_DIF + Size::NUMBER_LINE;
77 // If has temperature infos then pass Temp ASU 1, Temp ASU 2, Temp DIF
78 if(hasTemperature()) fshift += Size::TEMP_ASU1 + Size::TEMP_ASU2 + Size::TEMP_DIF;
79 // If has AnalogReadout pass them
80 if(hasAnalogReadout()) fshift = parseAnalogLine(fshift); // to be implemented
81 }
82 else
83 throw Exception(fmt::format("Version {} is not implemented", m_Version));
84
85 while(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::GLOBAL_TRAILER))
86 {
87 // If I found a FRAME_HEADER there is 2 cases :
88 // 1) Nothing inside so FRAME_TRAILER comes just after
89 // 2) Come MICROROC Header, BCID, DATA max 128 times
90 if(static_cast<std::uint8_t>(begin()[fshift]) == static_cast<std::uint8_t>(Value::FRAME_HEADER))
91 {
92 fshift += +Size::FRAME_HEADER;
93 if(static_cast<std::uint8_t>(begin()[fshift]) == static_cast<std::uint8_t>(Value::FRAME_TRAILER) || static_cast<std::uint8_t>(begin()[fshift]) == static_cast<std::uint8_t>(Value::FRAME_TRAILER_ERROR)) { fshift += +Size::FRAME_TRAILER; }
94 else
95 {
96 while(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::FRAME_TRAILER) && static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::FRAME_TRAILER_ERROR))
97 {
98 m_Frames.push_back(&begin()[fshift]);
99 fshift += Size::MICROROC_HEADER + Size::BCID + Size::DATA;
100 }
101 fshift += +Size::FRAME_TRAILER;
102 }
103 }
104 }
105 // Pass Global trailer
106 fshift += +Size::GLOBAL_TRAILER;
107 // Pass CRC MSB, CRC LSB
108 fshift += Size::CRC_MSB + Size::CRC_LSB;
109 theGetFramePtrReturn_ = fshift;
110}

122inline std::uint32_t Payload100::parseAnalogLine(const std::uint32_t& idx)
123{
124 std::uint32_t fshift{idx};
125 // Pass Header line
126 if(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::HEADER_LINE)) return fshift;
127 else
128 fshift += +Size::HEADER_LINE;
129 while(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::TRAILER_LINE))
130 {
131 m_Lines.push_back(&begin()[fshift]);
132 // Get Number of CHIPS
133 std::uint32_t nchip{begin()[fshift]};
134 // Pass Number of CHIPS, NB Asicline*64*16bits
135 fshift += +Size::NUMBER_CHIPS + static_cast<std::uint32_t>(Size::LINE_SIZE) * nchip;
136 }
137 // Pass Trailer line
138 fshift += +Size::TRAILER_LINE;
139 return fshift;
140}
