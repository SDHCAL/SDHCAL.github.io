27void RawdataReader::uncompress()
28{
29 static const std::size_t size_buffer{0x20000};
30 std::size_t shift{3 * sizeof(std::uint32_t) + sizeof(std::uint64_t)};
31 static bit8_t obuf[size_buffer];
32 unsigned long size_buffer_end{0x20000}; // NOLINT(runtime/int)
33 std::int8_t rc = ::uncompress(obuf, &size_buffer_end, &m_Buffer[shift], m_Buffer.size() - shift);
34 switch(rc)
35 {
36 case Z_OK: break;
37 case Z_MEM_ERROR: throw Exception(Z_MEM_ERROR, "Not enough memory"); break;
38 case Z_BUF_ERROR: throw Exception(Z_BUF_ERROR, "Not enough room in the output buffer"); break;
39 case Z_DATA_ERROR: throw Exception(Z_DATA_ERROR, "The input data was corrupted or incomplete"); break;
40 default: throw Exception("The input data was corrupted or incomplete"); break;
41 }
42 memcpy(&m_Buffer[shift], obuf, size_buffer_end);
43 m_Buffer.setSize(size_buffer_end + shift);
44}
