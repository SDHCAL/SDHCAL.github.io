63void RawBufferNavigator::setSCBuffer()
64{
65 if(!hasSlowControlData()) return;
66 if(m_SCbuffer.size() != 0) return; // deja fait
67 if(m_BadSCdata) return;
68 m_SCbuffer.set(&(getDIFBufferStart()[getEndOfDIFData()]));
69 // compute Slow Control size
70 std::size_t maxsize{m_Buffer.size() - m_DIFstartIndex - getEndOfDIFData() + 1}; // should I +1 here ?
71 uint32_t k{1}; // SC Header
72 uint32_t dif_ID{m_SCbuffer[1]};
73 uint32_t chipSize{m_SCbuffer[3]};
74 while((dif_ID != 0xa1 && m_SCbuffer[k] != 0xa1 && k < maxsize) || (dif_ID == 0xa1 && m_SCbuffer[k + 2] == chipSize && k < maxsize))
75 {
76 k += 2; // DIF ID + ASIC Header
77 uint32_t scsize = m_SCbuffer[k];
78 if(scsize != 74 && scsize != 109)
79 {
80 std::cout << "PROBLEM WITH SC SIZE " << scsize << std::endl;
81 k = 0;
82 m_BadSCdata = true;
83 break;
84 }
85 k++; // skip size bit
86 k += scsize; // skip the data
87 }
88 if(m_SCbuffer[k] == 0xa1 && !m_BadSCdata) m_SCbuffer.setSize(k + 1); // add the trailer
89 else
90 {
91 m_BadSCdata = true;
92 std::cout << "PROBLEM SC TRAILER NOT FOUND " << std::endl;
93 }
94}
