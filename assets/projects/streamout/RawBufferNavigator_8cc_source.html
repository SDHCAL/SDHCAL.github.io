85void RawBufferNavigator::setSCBuffer()
86{
87 if(!hasSlowControlData()) return;
88 if(m_SCbuffer.size() != 0) return; // deja fait
89 if(m_BadSCdata) return;
90 m_SCbuffer.set(&(getDIFBufferStart()[getEndOfDIFData()]));
91 // compute Slow Control size
92 std::size_t maxsize{m_Buffer.size() - m_StartPayload - getEndOfDIFData() + 1}; // should I +1 here ?
93 uint32_t k{1}; // SC Header
94 uint32_t dif_ID{m_SCbuffer[1]};
95 uint32_t chipSize{m_SCbuffer[3]};
96 while((dif_ID != 0xa1 && m_SCbuffer[k] != 0xa1 && k < maxsize) || (dif_ID == 0xa1 && m_SCbuffer[k + 2] == chipSize && k < maxsize))
97 {
98 k += 2; // DIF ID + ASIC Header
99 uint32_t scsize = m_SCbuffer[k];
100 if(scsize != 74 && scsize != 109)
101 {
102 m_Logger->error("PROBLEM WITH SC SIZE {}", scsize);
103 k = 0;
104 m_BadSCdata = true;
105 break;
106 }
107 k++; // skip size bit
108 k += scsize; // skip the data
109 }
110 if(m_SCbuffer[k] == 0xa1 && !m_BadSCdata) m_SCbuffer.setSize(k + 1); // add the trailer
111 else
112 {
113 m_BadSCdata = true;
114 m_Logger->error("PROBLEM SC TRAILER NOT FOUND ");
115 }
116}
