109 void setSCBuffer()
110 {
111 if(!hasSlowControlData()) return;
112 if(m_SCbuffer.size() != 0) return; // deja fait
113 if(m_BadSCdata) return;
114 m_SCbuffer.set(&(begin()[getEndOfDIFData()]));
115 // compute Slow Control size
116 std::size_t maxsize{size() - getEndOfDIFData() + 1}; // should I +1 here ?
117 uint32_t k{1}; // SC Header
118 uint32_t dif_ID{m_SCbuffer[1]};
119 uint32_t chipSize{m_SCbuffer[3]};
120 while((dif_ID != 0xa1 && m_SCbuffer[k] != 0xa1 && k < maxsize) || (dif_ID == 0xa1 && m_SCbuffer[k + 2] == chipSize && k < maxsize))
121 {
122 k += 2; // DIF ID + ASIC Header
123 uint32_t scsize = m_SCbuffer[k];
124 if(scsize != 74 && scsize != 109)
125 {
126 k = 0;
127 m_BadSCdata = true;
128 throw Exception(fmt::format("PROBLEM WITH SC SIZE {}", scsize));
129 }
130 k++; // skip size bit
131 k += scsize; // skip the data
132 }
133 if(m_SCbuffer[k] == 0xa1 && !m_BadSCdata) m_SCbuffer.setSize(k + 1); // add the trailer
134 else
135 {
136 m_BadSCdata = true;
137 throw Exception(fmt::format("PROBLEM SC TRAILER NOT FOUND "));
138 }
139 }

217inline std::uint32_t PayloadParser::getFramePtr()
218{
219 std::uint32_t fshift{0};
220 if(m_Version >= 13)
221 {
222 fshift = DU::LINES_SHIFT + 1;
223 if(hasTemperature()) fshift = DU::TDIF_SHIFT + 1; // jenlev 1
224 if(hasAnalogReadout()) fshift = getAnalogPtr(fshift); // to be implemented
225 }
226 else
227 fshift = DU::BCID_SHIFT + 3;
228 if(begin()[fshift] != DU::START_OF_FRAME) { throw Exception(fmt::format("This is not a start of frame {}", to_hex(begin()[fshift]))); }
229 do {
230 if(begin()[fshift] == DU::END_OF_DIF) return fshift;
231 if(begin()[fshift] == DU::START_OF_FRAME) fshift++;
232 if(begin()[fshift] == DU::END_OF_FRAME)
233 {
234 fshift++;
235 continue;
236 }
237 std::uint32_t header = getFrameAsicHeaderInternal(&begin()[fshift]);
238 if(header == DU::END_OF_FRAME) return (fshift + 2);
239 if(header < 1 || header > 48) { throw Exception(fmt::format("{} Header problem {}", header, fshift)); }
240 theFrames_.push_back(&begin()[fshift]);
241 fshift += DU::FRAME_SIZE;
242 if(fshift > size()) { throw Exception(fmt::format("fshift {} exceed {}", fshift, size())); }
243 if(begin()[fshift] == DU::END_OF_FRAME) fshift++;
244 } while(true);
245}

248{

252 while(begin()[fshift] != DU::END_OF_LINES)

255 std::uint32_t nchip{begin()[fshift]};
256 fshift += 1 + nchip * 64 * 2;
257 }
258 return fshift++;
