145inline std::uint32_t DIFPtr::getFramePtr()
146{
147 std::uint32_t fshift{0};
148 if(DATA_FORMAT_VERSION >= 13)
149 {
150 fshift = DU::LINES_SHIFT + 1;
151 if(hasTemperature()) fshift = DU::TDIF_SHIFT + 1; // jenlev 1
152 if(hasAnalogReadout()) fshift = getAnalogPtr(fshift); // to be implemented
153 }
154 else
155 fshift = DU::BCID_SHIFT + 3;
156 if(theDIF_[fshift] != DU::START_OF_FRAME)
157 {
158 std::cout << "This is not a start of frame " << to_hex(theDIF_[fshift]) << " \n";
159 return fshift;
160 }
161 do {
162 if(theDIF_[fshift] == DU::END_OF_DIF) return fshift;
163 if(theDIF_[fshift] == DU::START_OF_FRAME) fshift++;
164 if(theDIF_[fshift] == DU::END_OF_FRAME)
165 {
166 fshift++;
167 continue;
168 }
169 std::uint32_t header = getFrameAsicHeaderInternal(&theDIF_[fshift]);
170 if(header == DU::END_OF_FRAME) return (fshift + 2);
171 if(header < 1 || header > 48) { throw header + " Header problem " + fshift; }
172 theFrames_.push_back(&theDIF_[fshift]);
173 fshift += DU::FRAME_SIZE;
174 if(fshift > theSize_)
175 {
176 std::cout << "fshift " << fshift << " exceed " << theSize_ << "\n";
177 return fshift;
178 }
179 if(theDIF_[fshift] == DU::END_OF_FRAME) fshift++;
180 } while(true);
181}

183inline std::uint32_t DIFPtr::getAnalogPtr(const std::uint32_t& idx)
184{
185 std::uint32_t fshift{idx};
186 if(theDIF_[fshift] != DU::START_OF_LINES) return fshift;
187 fshift++;
188 while(theDIF_[fshift] != DU::END_OF_LINES)
189 {
190 theLines_.push_back(&theDIF_[fshift]);
191 std::uint32_t nchip{theDIF_[fshift]};
192 fshift += 1 + nchip * 64 * 2;
193 }
194 return fshift++;
195}
