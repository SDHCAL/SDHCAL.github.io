111 void setSCBuffer()
112 {
113 if(!hasSlowControlData()) return;
114 if(m_SCbuffer.size() != 0) return; // deja fait
115 if(m_BadSCdata) return;
116 m_SCbuffer.set(&(theDIF_[getEndOfDIFData()]));
117 // compute Slow Control size
118 std::size_t maxsize{theSize_ - getEndOfDIFData() + 1}; // should I +1 here ?
119 uint32_t k{1}; // SC Header
120 uint32_t dif_ID{m_SCbuffer[1]};
121 uint32_t chipSize{m_SCbuffer[3]};
122 while((dif_ID != 0xa1 && m_SCbuffer[k] != 0xa1 && k < maxsize) || (dif_ID == 0xa1 && m_SCbuffer[k + 2] == chipSize && k < maxsize))
123 {
124 k += 2; // DIF ID + ASIC Header
125 uint32_t scsize = m_SCbuffer[k];
126 if(scsize != 74 && scsize != 109)
127 {
128 k = 0;
129 m_BadSCdata = true;
130 throw Exception(fmt::format("PROBLEM WITH SC SIZE {}", scsize));
131 }
132 k++; // skip size bit
133 k += scsize; // skip the data
134 }
135 if(m_SCbuffer[k] == 0xa1 && !m_BadSCdata) m_SCbuffer.setSize(k + 1); // add the trailer
136 else
137 {
138 m_BadSCdata = true;
139 throw Exception(fmt::format("PROBLEM SC TRAILER NOT FOUND "));
140 }

231inline std::uint32_t DIFPtr::getFramePtr()
232{
233 std::uint32_t fshift{0};
234 if(DATA_FORMAT_VERSION >= 13)
235 {
236 fshift = DU::LINES_SHIFT + 1;
237 if(hasTemperature()) fshift = DU::TDIF_SHIFT + 1; // jenlev 1
238 if(hasAnalogReadout()) fshift = getAnalogPtr(fshift); // to be implemented
239 }
240 else
241 fshift = DU::BCID_SHIFT + 3;
242 if(theDIF_[fshift] != DU::START_OF_FRAME) { throw Exception(fmt::format("This is not a start of frame {}", to_hex(theDIF_[fshift]))); }
243 do {
244 if(theDIF_[fshift] == DU::END_OF_DIF) return fshift;
245 if(theDIF_[fshift] == DU::START_OF_FRAME) fshift++;
246 if(theDIF_[fshift] == DU::END_OF_FRAME)
247 {
248 fshift++;
249 continue;
250 }
251 std::uint32_t header = getFrameAsicHeaderInternal(&theDIF_[fshift]);
252 if(header == DU::END_OF_FRAME) return (fshift + 2);
253 if(header < 1 || header > 48) { throw Exception(fmt::format("{} Header problem {}", header, fshift)); }
254 theFrames_.push_back(&theDIF_[fshift]);
255 fshift += DU::FRAME_SIZE;
256 if(fshift > theSize_) { throw Exception(fmt::format("fshift {} exceed {}", fshift, theSize_)); }
257 if(theDIF_[fshift] == DU::END_OF_FRAME) fshift++;
258 } while(true);
259}

261inline std::uint32_t DIFPtr::getAnalogPtr(const std::uint32_t& idx)
262{
263 std::uint32_t fshift{idx};
264 if(theDIF_[fshift] != DU::START_OF_LINES) return fshift;
265 fshift++;
266 while(theDIF_[fshift] != DU::END_OF_LINES)
267 {
268 theLines_.push_back(&theDIF_[fshift]);
269 std::uint32_t nchip{theDIF_[fshift]};
270 fshift += 1 + nchip * 64 * 2;
271 }
272 return fshift++;
