100inline std::uint32_t Payload100::parsePayload()
101{
102 std::uint32_t fshift{static_cast<std::uint32_t>(Size::GLOBAL_HEADER)}; // Pass Global Header
103 if(m_Version >= 13)
104 {
105 // Pass DIF_ID, DIF Trigger counter, Information counter, Global Trigger counter, Absolute BCID, BCID DIF, NB line
106 fshift += Size::DIF_IF + Size::DIF_TRIGGER_COUNTER + Size::INFORMATION_COUNTER + Size::GLOBAL_TRIGGER_COUNTER + Size::ABSOLUTE_BCID + Size::BCID_DIF + Size::NUMBER_LINE;
107 // If has temperature infos then pass Temp ASU 1, Temp ASU 2, Temp DIF
108 if(hasTemperature()) fshift += Size::TEMP_ASU1 + Size::TEMP_ASU2 + Size::TEMP_DIF;
109 // If has AnalogReadout pass them
110 if(hasAnalogReadout()) fshift = parseAnalogLine(fshift); // to be implemented
111 }
112 else
113 throw Exception(fmt::format("Version {} is not implemented", m_Version));
114
115 while(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::GLOBAL_TRAILER))
116 {
117 // If I found a FRAME_HEADER there is 2 cases :
118 // 1) Nothing inside so FRAME_TRAILER comes just after
119 // 2) Come MICROROC Header, BCID, DATA max 128 times
120 if(static_cast<std::uint8_t>(begin()[fshift]) == static_cast<std::uint8_t>(Value::FRAME_HEADER))
121 {
122 fshift += +Size::FRAME_HEADER;
123 if(static_cast<std::uint8_t>(begin()[fshift]) == static_cast<std::uint8_t>(Value::FRAME_TRAILER) || static_cast<std::uint8_t>(begin()[fshift]) == static_cast<std::uint8_t>(Value::FRAME_TRAILER_ERROR)) { fshift += +Size::FRAME_TRAILER; }
124 else
125 {
126 while(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::FRAME_TRAILER) && static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::FRAME_TRAILER_ERROR))
127 {
128 m_Frames.push_back(&begin()[fshift]);
129 fshift += Size::MICROROC_HEADER + Size::BCID + Size::DATA;
130 }
131 fshift += +Size::FRAME_TRAILER;
132 }
133 }
134 }
135 // Pass Global trailer
136 fshift += +Size::GLOBAL_TRAILER;
137 // Pass CRC MSB, CRC LSB
138 fshift += Size::CRC_MSB + Size::CRC_LSB;
139 return fshift;
140}

152inline std::uint32_t Payload100::parseAnalogLine(const std::uint32_t& idx)
153{
154 std::uint32_t fshift{idx};
155 // Pass Header line
156 if(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::HEADER_LINE)) return fshift;
157 else
158 fshift += +Size::HEADER_LINE;
159 while(static_cast<std::uint8_t>(begin()[fshift]) != static_cast<std::uint8_t>(Value::TRAILER_LINE))
160 {
161 m_Lines.push_back(&begin()[fshift]);
162 // Get Number of CHIPS
163 std::uint32_t nchip{begin()[fshift]};
164 // Pass Number of CHIPS, NB Asicline*64*16bits
165 fshift += +Size::NUMBER_CHIPS + static_cast<std::uint32_t>(Size::LINE_SIZE) * nchip;
166 }
167 // Pass Trailer line
168 fshift += +Size::TRAILER_LINE;
169 return fshift;
170}
