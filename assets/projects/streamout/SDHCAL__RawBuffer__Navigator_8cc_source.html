68void SDHCAL_RawBuffer_Navigator::setSCBuffer()
69{
70 if(!hasSlowControlData()) return;
71 if(m_SCbuffer.size() != 0) return; // deja fait
72 if(m_BadSCdata) return;
73 m_SCbuffer.set(&(getDIFBufferStart()[getEndOfDIFData()]));
74 // compute Slow Control size
75 std::size_t maxsize{m_Buffer.size() - m_DIFstartIndex - getEndOfDIFData() + 1}; // should I +1 here ?
76 uint32_t k{1}; // SC Header
77 uint32_t dif_ID{m_SCbuffer[1]};
78 uint32_t chipSize{m_SCbuffer[3]};
79 while((dif_ID != 0xa1 && m_SCbuffer[k] != 0xa1 && k < maxsize) || (dif_ID == 0xa1 && m_SCbuffer[k + 2] == chipSize && k < maxsize))
80 {
81 k += 2; // DIF ID + ASIC Header
82 uint32_t scsize = m_SCbuffer[k];
83 if(scsize != 74 && scsize != 109)
84 {
85 std::cout << "PROBLEM WITH SC SIZE " << scsize << std::endl;
86 k = 0;
87 m_BadSCdata = true;
88 break;
89 }
90 k++; // skip size bit
91 k += scsize; // skip the data
92 }
93 if(m_SCbuffer[k] == 0xa1 && !m_BadSCdata) m_SCbuffer.setSize(k + 1); // add the trailer
94 else
95 {
96 m_BadSCdata = true;
97 std::cout << "PROBLEM SC TRAILER NOT FOUND " << std::endl;
98 }
99}
